function goToDeck(di, si) {

currentDeck = di;

currentSlide = si;

renderState();

}


function renderState() {

document.getElementById('overview').classList.toggle('visible', showingOverview);

document.querySelectorAll('.slide').forEach(s => {

const active = !showingOverview && parseInt(s.dataset.deck) === currentDeck && parseInt(s.dataset.slide) === currentSlide;

s.classList.toggle('visible', active);

});

document.querySelectorAll('.deck-item').forEach((el, i) => el.classList.toggle('active', !showingOverview && i === currentDeck));


const total = DECKS[currentDeck]?.slides.length || 1;

document.getElementById('cur-num').textContent = showingOverview ? '—' : currentSlide + 1;

document.getElementById('total-num').textContent = showingOverview ? '—' : total;

document.getElementById('progress-fill').style.width = showingOverview ? '0%' : ((currentSlide + 1) / total * 100) + '%';

document.getElementById('btn-prev').disabled = showingOverview || (currentDeck === 0 && currentSlide === 0);

document.getElementById('btn-next').disabled = showingOverview || (currentDeck === DECKS.length - 1 && currentSlide === DECKS[currentDeck].slides.length - 1);

}


document.getElementById('btn-prev').addEventListener('click', () => {

if (showingOverview) return;

if (currentSlide > 0) currentSlide--;

else if (currentDeck > 0) { currentDeck--; currentSlide = DECKS[currentDeck].slides.length - 1; }

renderState();

});

document.getElementById('btn-next').addEventListener('click', () => {

if (showingOverview) return;

if (currentSlide < DECKS[currentDeck].slides.length - 1) currentSlide++;

else if (currentDeck < DECKS.length - 1) { currentDeck++; currentSlide = 0; }

renderState();

});

document.getElementById('overview-btn').addEventListener('click', () => { showingOverview = true; renderState(); });


document.addEventListener('keydown', e => {

if (showingOverview) return;

if (e.key === 'ArrowRight' || e.key === 'ArrowDown') document.getElementById('btn-next').click();

if (e.key === 'ArrowLeft' || e.key === 'ArrowUp') document.getElementById('btn-prev').click();

if (e.key === 'Escape') { showingOverview = true; renderState(); }

});


renderState();
